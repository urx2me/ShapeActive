    Private Sub LoadUserProfile()
        Dim connectionString As String = "server=localhost;user id=root;password=nics1108;database=shapeactive"
        Dim query As String = "SELECT dname, demail, dpassword, dage, dheight, dcurrentWeight, dbmi FROM tblusers WHERE duserId = @userId"

    ' Display BMI on the form
    ' Calculate BMI based on age, height, and weight
    Private Function CalculateBMI(ByVal age As Double, ByVal height As Double, ByVal weight As Double) As Double
        ' BMI formula: BMI = weight (kg) / (height (m))^2
        ' Convert height from ft to m
        Dim heightInMeters As Double = height * 0.3048
        ' Calculate BMI
        Return weight / (heightInMeters * heightInMeters)
    End Function

    ' Display BMI on the form
    Private Sub CalculateAndDisplayBMI(ByVal age As Object, ByVal height As Object, ByVal weight As Object)
        If age IsNot DBNull.Value AndAlso height IsNot DBNull.Value AndAlso weight IsNot DBNull.Value Then
            Dim bmi As Double = CalculateBMI(Convert.ToDouble(age), Convert.ToDouble(height), Convert.ToDouble(weight))
            txtBMI.Text = $"{bmi:F2}"

            ' Customize the BMI result based on age
            Dim bmiResult As String = GetBMIResult(bmi, Convert.ToInt32(age))
            lblBMIResult.Text = $"Your BMI Result: {bmiResult}"
        Else
            txtBMI.Text = "BMI: N/A"
            lblBMIResult.Text = "Your BMI Result: N/A"
        End If
    End Sub
